📝 3. Form Components

Tất cả form components đều tích hợp với Formik và FastField để tối ưu hiệu năng.

📄 CommonTextField Form
Input cơ bản với hỗ trợ debounce, auto capitalize, auto replace text. Tích hợp FastField để tối ưu re-render.
Props:
name: string * • label: string • required: boolean • disabled: boolean • readOnly: boolean • fullWidth: boolean (default: true) • size: 'small' | 'medium' (default: 'small') • debounceTime: number (default: 300) • autoCapitalize: boolean • autoReplaceText: boolean • isTextArea: boolean • rows: number (default: 1)
📄 CommonSelectInput Form
Select với Autocomplete từ MUI. Hỗ trợ search/filter, disable clear, custom display.
Props:
name: string * • options: Array * • keyValue: string (default: 'value') • displayValue: string (default: 'name') • noNullOption: boolean • multiple: boolean
✅ CommonCheckBox Form
Checkbox hỗ trợ giá trị null, align trái/phải, custom size và màu sắc.
Props:
name: string * • label: string • value: any (default: true) • isNoneValue: boolean • align: 'left' | 'right' (default: 'right') • size: 'small' | 'medium'
📅 CommonDateTimePicker Form
Date/Time Picker từ MUI X Date Pickers. Hỗ trợ Date, Time, DateTime, disable past/future.
Props:
name: string * • format: string (default: 'dd/MM/yyyy') • isDateTimePicker: boolean • isTimePicker: boolean • disablePast: boolean • disableFuture: boolean • minDate: Date • maxDate: Date
🔍 CommonPagingAutocomplete Form
Autocomplete phân trang với infinite scroll, search debounce, hỗ trợ multiple, select all.
Props:
name: string * • api: function * • searchObject: object • displayLabel: string (default: 'name') • displayValue: string (default: 'id') • multiple: boolean • enableSelectAll: boolean • keepOpenOnSelection: boolean
🔒 CommonPasswordInput Form
Password input với toggle show/hide password, tích hợp Formik.
Props:
name: string * • label: string • required: boolean • description: string
⭐ RequiredLabel Form
Helper component hiển thị label với dấu * khi required.
Props:
label: string * • required: boolean
❌ ErrorIcon Form
Hiển thị icon error với tooltip helper text, dùng khi fullWidth=false.
Props:
helperText: string *
🎯 FormikFocusError Form
Tự động focus vào field có lỗi khi submit form. Kết hợp với toast warning.
Props:
formId: string • showOther: boolean • setShowOther: function • errorsItem: object

📝 Cấu trúc store mẫu:
// stores/memberStore.js
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

const useMemberStore = create(
  (set, get) => ({
    // ===== STATE =====
    members: [],
    selectedMember: null,
    totalElements: 0,
    loading: false,
    error: null,
    filters: {
      pageIndex: 1,
      pageSize: 10,
      keyword: '',
      generation: '',
    },

    // ===== ACTIONS =====
    fetchMembers: async (params = {}) => {
      set({ loading: true, error: null });
      try {
        const response = await memberService.paging({ ...get().filters, ...params });
        set({
          members: response.data.content,
          totalElements: response.data.totalElements,
          loading: false,
        });
      } catch (error) {
        set({ loading: false, error: error.message });
      }
    },

    setFilters: (filters) => {
      set((state) => ({
        filters: { ...state.filters, ...filters, pageIndex: 1 }
      }));
      get().fetchMembers();
    },

    // ... các actions khác
  }
);

export default useMemberStore;

🌐 10. Service Pattern

📝 Cấu trúc service mẫu:
// modules/members/services/memberService.js
import api from '../../../services/api';

const API_PATH = import.meta.env.VITE_API_URL + '/api/profiles';

export const pagingMembers = (dto) => {
  return api.post(`${API_PATH}/page`, dto);
};

export const getMemberById = (id) => {
  return api.get(`${API_PATH}/${id}`);
};

export const saveMember = (dto) => {
  return api.post(`${API_PATH}/save`, dto);
};

export const deleteMember = (id) => {
  return api.delete(`${API_PATH}/${id}`);
};

export const deleteMultipleMembers = (ids) => {
  return api.post(`${API_PATH}/deleteMultiple`, ids);
};
📝 API Interceptor (services/api.js):
import axios from 'axios';

const api = axios.create({
  baseURL: import.meta.env.VITE_API_URL,
  timeout: 30000,
});

// Request interceptor - gắn JWT token
api.interceptors.request.use(
  (config) => {
    const token = localStorage.getItem('accessToken');
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => Promise.reject(error)
});

// Response interceptor - refresh token
api.interceptors.response.use(
  (response) => response,
  async (error) => {
    if (error.response?.status === 401) {
      // Handle token refresh
    }
    return Promise.reject(error);
  }
);

export default api;
